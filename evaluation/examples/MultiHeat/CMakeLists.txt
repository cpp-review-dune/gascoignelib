set(MULTIHEAT_EXEC MultiHeat)

FILE(GLOB_RECURSE MULTIHEAT_SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)
ADD_EXECUTABLE(${MULTIHEAT_EXEC} ${MULTIHEAT_SOURCE})

set(HEAT_COMP 3 CACHE STRING "multiheat components")
LIST(APPEND GASCOIGNE_DEFINITIONS -DHEAT_COMP=${HEAT_COMP})

target_compile_definitions(${MULTIHEAT_EXEC} PUBLIC
	${GASCOIGNE_DEFINITIONS}
)

target_include_directories(${MULTIHEAT_EXEC} PUBLIC
	${GASCOIGNE_INCLUDE_DIRS}
	${CMAKE_CURRENT_SOURCE_DIR}
)

target_link_libraries(${MULTIHEAT_EXEC} 
	GascoigneStd
)

set(ALL_CXX_SOURCE_FILES ${ALL_CXX_SOURCE_FILES} ${MULTIHEAT_SOURCE} PARENT_SCOPE)
